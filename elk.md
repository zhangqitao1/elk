安装 使用docker-compose
======================

#docker-compose.yml
#docker-compose up -d

```
version: '3'
services:

  elasticsearch:
    image: elastic/elasticsearch:7.17.3
    environment: 
      - "ES_JAVA_OPTS=-Xms256m -Xmx256m"
      - discovery.type=single-node 
    ports:
      - "9200:9200"
      - "9300:9300"


    ulimits:
      memlock:
        soft: -1
        hard: -1
      nofile:
        soft: 65536
        hard: 65536
  kibana:
    image: elastic/kibana:7.17.3
    ports:
      - "5601:5601"
      
      
es地址：http://127.0.0.1:9200/    
kibana：http://localhost:5601/
```
